 \begin{mytikzbox}{ טריגונומטריה}{
\textbf{סכום זוויות:}
\begin{gather*}\sin(a\pm b) = \sin(a)\cos(b)\pm \cos(a)\sin(b) \\
\cos(a\pm b)=\cos(a)\cos(b)\mp \sin(a)\sin(b)\end{gather*}

\textbf{מכפלת פונקציות:}
\begin{gather*}\cos(a)\cos(b)=\frac{1}{2}\left[ \cos(a+ b)+\cos(a-b) \right] \\
\sin(a)\sin(b)=\frac{1}{2}[\cos(a- b)-\cos(a+b)] \\
\cos(a)\sin(b)=\frac{1}{2}[\sin(a+ b)+\sin(a-b)]\end{gather*}

\textbf{בעזרת מרוכבים}
\begin{gather*}
    \sin(z) = \frac{e^{iz}-e^{-iz}}{2i}\qquad \cos(z) = \frac{e^{iz}+e^{-iz}}{2}\\
    \cosh(x)=\cos(ix)\qquad \sinh(x)=-i\sin(ix)
\end{gather*}

 \textbf{זהיות זווית כפולה}
$$\cos(2x)=\cos^2 (x) - \sin^2 (x)\quad \sin (2x) = 2\sin (x)\cos(x)$$

\textbf{זהיות חצי זווית}
 \begin{gather*}\sin\left(\frac \theta2\right)=\pm\sqrt{\frac{1-\cos \theta}2}\quad\cos\left(\frac \theta2\right)=\pm\sqrt{\frac{1+\cos \theta}2}\\
 \tan\left(\frac \theta2\right)=\pm\sqrt{\frac{1-\cos \theta}{1+\cos \theta}}=\frac{\sin \theta}{1+\cos \theta}=\frac{1-\cos \theta}{\sin \theta}
 \end{gather*}
 \vspace{-10pt}

    \begin{mytikzbox}{קואורדינטות ספריות}{
    \textbf{פרמטריזציה של קואורדינטות ספריות:}
$$
    x = r \sin\theta \cos\phi \qquad
    y = r \sin\theta \sin\phi \qquad
    z = r \cos\theta $$ 
    \textbf{טווחי משתנים:}
    \[
    r \geq 0,\quad 0 \leq \theta \leq \pi,\quad 0 \leq \phi < 2\pi
    \]

    \textbf{אלמנט נפח:}
    \[
    \mathrm{d}^3\mathbf{r} = r^2 \sin\theta\, \mathrm{d}r\, \mathrm{d}\theta\, \mathrm{d}\phi
    \]
    }{0.46}
