    \subsection*{מערכות עם אינטראקציה}\unskip
    \begin{mytikzbox}{מודל איזינג חד מימדי מחזורי}{
\unskip עבור מערכות עם אינטראקציה: $H(\text{micro})=\sum_{i=1}^{N} H_{\text{free}}^{i}+H_{\text{int}}$. 

\textbf{מודל איזינג חד מימדי מחזורי:}
$$H_{\text{free}}^{i}=\sum_{i=1}^N s_i\qquad H_{\text{int}}^{(\text{micro})}=-J\sum_{i=1}^{N} s_{i}s_{i+1}\qquad s_{N+1}\equiv s_1$$
    
\textbf{פונקציית החלוקה:}
ניתן לכתוב את פונקציית החלוקה על ידי:
$$Z_{G}=\mathrm{tr}\left( \exp \left\{  \beta\left( J\sum_{i=1}^{N} s_{i}s_{i+1} + B\sum_{i=1}^{N} s_{i} \right)  \right\} \right)$$
$$
Z_G = \mathrm{tr} \prod_{i=1}^N \exp\left\{ \beta \left[ J s_i s_{i+1} + \frac{B}{2}(s_i + s_{i+1}) \right] \right\}
$$
פונקציית החלוקה ניתנת לכתיבה בעזרת מטריצת מעבר $M$:
$$
Z_{G}=\sum_{s_{1}}(M^{N})_{s_{1}s_{1}}=\mathrm{Tr}(M^{N})$$\\[-10pt]
כאשר $M_{ab} = \exp\left(\beta J ab + \frac{\beta B}{2}(a+b)\right)$ ולספין חצי:
$$
M = \begin{pmatrix}
e^{\beta(J-B)} & e^{-\beta J} \\
e^{-\beta J} & e^{\beta(J+B)}
\end{pmatrix}
$$
וניתן למצוא ע"ע $\lambda_{\pm}$ כך ש-$Z_{G}=\mathrm{Tr}(M_{d}^{N})=\lambda_{+}^{N}+\lambda_{-}^{N}$

\textbf{אנרגיה חופשית של גיבס:}
$$G=-{\frac{1}{\beta}}\log Z_{G}=-{\frac{1}{\beta}}\log\left(\lambda_{+}^{N}+\lambda_{-}^{N}\right)\xrightarrow{N\to\infty}-{\frac{N}{\beta}}\log\lambda_{+}$$

\textbf{ערכים עצמיים:}
$$\lambda_{\pm}=e^{\beta J}\cosh(\beta B)\pm\sqrt{e^{2\beta J}\sinh^{2}(\beta B)+e^{-2\beta J}}$$

\textbf{קורולציה:}
\textbf{קורלציה בין ספינים:}
\[
\langle s_j s_k \rangle = \frac{1}{Z_G} \mathrm{Tr}\left[ \sigma_z M^{k-j} \sigma_z M^{N-(k-j)} \right]\quad \sigma_z = \begin{pmatrix} 1 & 0 \\ 0 & -1 \end{pmatrix}
\]

נובע
הגדרת התוחלת - $
\langle s_j s_k \rangle = \frac{1}{Z_G} \mathrm{tr}\left( s_j s_k e^{-\beta H} \right)$
פונקציית החלוקה נכתבת כמכפלת מטריצות מעבר, ולכן אפשר לייצג את $s_j, s_k$ בעזרת $\sigma_z$ במיקום המתאים.

\textbf{תוצאה עבור ערכים עצמיים:}
אם $\lambda_+, \lambda_-$ הם הערכים העצמיים של $M$:
\[
\langle s_j s_k \rangle = \left( \frac{\lambda_-}{\lambda_+} \right)^{k-j} + \left( \frac{\lambda_-}{\lambda_+} \right)^{N-(k-j)}
\]

\textbf{קירוב למרחקים קטנים (אורך קורלציה):}
כאשר $k-j \ll N$ (כלומר, המרחק בין הספינים קטן יחסית לאורך השרשרת), אפשר לקרב את הקורלציה בין ספינים כך:
\[
\langle s_j s_k \rangle \approx \left[ \tanh(\beta J) \right]^{k-j}
\]

\textbf{אורך קורלציה:}
בפרט, עבור טמפרטורות נמוכות (כלומר, $\beta J \gg 1$), הקורלציה דועכת בצורה מעריכית עם המרחק:
\[
\langle s_j s_k \rangle \sim e^{-(k-j)/\xi}\qquad \xi = -\frac{1}{\log \tanh(\beta J)}
\]
כאשר $\xi$ הוא אורך הקורלציה. עבור טמפרטורה נמוכה במיוחד, מתקבל בקירוב 
$\xi \approx \frac{1}{2} e^{2\beta J}$.
}{0.46}\end{mytikzbox}
\begin{mytikzbox}{קירוב שדה ממוצע - מודל איזינג}{
מערכת עם אינטראקציה של שכנים קרובים, כלומר מהצורה ${H}_{\mathrm{int}}=-J\sum_{\langle i,j\rangle}s_{i}s_{j}$.
נגדיר את המרחק מהממוצע על ידי $\delta s_{i}=s_{i}-m$. נזמן ב-$d$ את המימד וב-$z=2d$ את דרגת החופש של כל קודקוד (מספר השכנים הקרובים).

$$Z_{G}=\mathrm{tr}\left( \exp\left\{ \beta\left( J\sum_{\langle i,j\rangle}s_{i}s_{j} + B\sum_{i}s_{i} \right) \right\} \right)$$
$$\!\!\!\!\!=\!\mathrm{tr}\exp\!\left\{\!\!J\beta\!\sum_{\langle i,j\rangle}\!\left(\delta s_{i}\delta s_{j}\!+\!m\left(\delta s_{i}\!+\!\delta s_{j}\right)\!\!+\!\!m^{2}\right)\!\!+\!\!\beta B\sum_{i}\!\delta s_{i}\!\!+\!\!\beta B N m\!\!\right\}$$

$$Z_{G}^{\mathsf{M F}}=\exp\left\{-{\frac{\beta J N z m^{2}}{2}}\right\}\cdot(2\cosh\left(\beta B_{\mathsf{M F}}\right))^{N}$$

משוואת משוואה עצמית למגנטיזציה:

\begin{mytikzbox}{טמפרטורה קריטית ומעבר פאזה במודל איזינג}{
נסתכל על משוואת המגנטיזציה בשדה אפס ($B=0$):
\[
m = \tanh(\beta J z m)
\]
קיים טמפרטורה קריטית $T_c = \frac{Jz}{k_B}$ כך שאם $T > T_c$ הפתרון היחיד הוא $m=0$, ואם $T < T_c$ יש פתרון נוסף $m \neq 0$.

\textbf{הוכחה:} נפתח את הטנגנס היפרבולי בטור טיילור:
\[\!\!\!\!\!
m \!\approx\! \beta J z m - \frac{(\beta J z m)^3}{3} + \cdots
\!\Rightarrow\!
m \!\left[ \!(\beta J z \!-\! 1) \!- \!\frac{(\beta J z)^3 m^2}{3} \!+\! \cdots \right] \!=\! 0
\]
לכן $m=0$ תמיד פתרון, ויש פתרון נוסף אם ורק אם:
\[
\beta J z - 1 > 0 \implies T < T_c = \frac{Jz}{k_B}
\]
 מעבר הפאזה מתרחש כאשר מתקבל פתרון לא טריוויאלי $m \neq 0$.

\textbf{הסוספטביליות:}
\[
\chi = \left. \frac{\partial m(B)}{\partial B} \right|_{B=0}=\beta \frac{1 + Jz\chi}{\cosh^2(Jz m \beta)}
= \frac{\beta}{\cosh^2(Jz m \beta) - Jz\beta}
\]

\textbf{טענה:} עבור $T<T_c$:
\[
m = \sqrt{ 3 \left( \frac{T}{T_c} \right)^2 \frac{T_c - T}{T_c} }
\]
ועבור הסוספטביליות:
\[
\chi\big|_{T\to T_c^-} = \frac{1}{2k_B (T_c - T)} + O((T_c - T)^0)
\]
ובכתיבה חזקתית: $\chi \sim (T_c - T)^{-\gamma}$, כאשר כאן $\gamma_{MF} = 1$.
}{0.46}\end{mytikzbox}

\[

\begin{enumerate}
    \item \textbf{כתיבת האינטגרל:}\\
    נרצה לחשב אינטגרל מהצורה:
    \[

\!\!\!\!\!=\!\!\! \int \!\!\mathcal{D}\lambda\, \exp\!\!\left(\!-\frac{1}{2}\!\!\int\limits_0^T \!\!\lambda^2(t)g^2(v(t),t)\,\mathrm{d}t \!+\!i\!\!\int\limits_0^T \!\!\lambda(t)(\dot{v}(t)\!-\!f(v(t),t))\,\mathrm{d}t\right)
\]

\textbf{תוצאה סופית:}
\[
P[v(t)] \propto \exp\left(-\frac{1}{2}\int_0^dT \left(\frac{\dot{v}(t)-f(v(t),t)}{g(v(t),t)}\right)^2 \mathrm{d}t\right)
\]
}{0.46}
\end{mytikzbox}
\begin{mytikzbox}{פיתוח קרמר-מוייל}{
\textbf{פיתוח Kramers–Moyal} מתאר את האבולוציה של צפיפות ההסתברות $P_v^t(v)$ עבור תהליך מרקובי רציף בזמן:

\[
\frac{\partial P_v^t(v)}{\partial t} = \sum_{n=1}^{\infty} \frac{(-1)^n}{n!} \cdot \frac{\partial^n}{\partial v^n} \left[ D_n(v,t)\,P_v^t(v) \right]
\]

כאשר המקדמים $D_n(v,t)$ מוגדרים על ידי:

\[
D_n(v,t) = \lim_{\Delta t \to 0} \frac{ \left\langle \left( V(t+\Delta t) - v \right)^n \right\rangle_{V(t)=v} }{ \Delta t }
\]

כלומר, $D_n$ הוא רגע השינוי ה-$n$ של $v$ ליחידת זמן, בתנאי שהמערכת הייתה ב-$v$ בזמן $t$.

\textbf{מקרה פרטי: משוואת פוקר פלאנק}  
אם מתקיים $D_n = 0$ לכל $n > 2$, נקבל את משוואת פוקר פלאנק:

\[
\frac{\partial P_v^t(v)}{\partial t} = 
- \frac{\partial}{\partial v} \left[ D_1(v,t)\,P_v^t(v) \right] 
+ \frac{1}{2} \cdot \frac{\partial^2}{\partial v^2} \left[ D_2(v,t)\,P_v^t(v) \right]
\]

כאשר:
\[\!\!
D_1 = \lim_{\Delta t \to 0} \frac{ \left\langle V(t+\Delta t) - v \right\rangle }{ \Delta t }
\quad
D_2 = \lim_{\Delta t \to 0} \frac{ \left\langle \left( V(t+\Delta t) - v \right)^2 \right\rangle }{ \Delta t }
\]
}{0.46}
\end{mytikzbox}
\begin{mytikzbox}{פוקר-פלאנק עבור לנג'בן כללי}{
\textbf{משוואת לנג'בן:}
\[
\gamma \frac{dx}{dt} = F(x) + \eta(t)
\qquad
\langle \eta(t) \rangle = 0,\;
\langle \eta(t)\eta(t') \rangle = \Lambda \delta(t-t')
\]

\textbf{פוקר-פלאנק:}
\[
\frac{\partial P}{\partial t} = \frac{1}{2} \frac{\Lambda}{\gamma^2} \frac{\partial^2 P}{\partial x^2}
- \frac{1}{\gamma} \frac{\partial}{\partial x} [F(x) P]
\]

\textbf{ריבוי משתנים:}
\[
\gamma_i \frac{dx_i}{dt} = F_i(\vec{x}) + \eta_i(t)
\qquad
\langle \eta_i(t)\eta_j(t') \rangle = \Lambda_i \delta_{ij} \delta(t-t')
\]
\[
\frac{\partial P}{\partial t} = \sum_{i=1}^N \left[
\frac{1}{2} \frac{\Lambda_i}{\gamma_i^2} \frac{\partial^2 P}{\partial x_i^2}
- \frac{1}{\gamma_i} \frac{\partial}{\partial x_i} (F_i P)
\right]
\]

\textbf{אבולוציה של ממוצע:}
\[
\frac{d}{dt} \langle A(x) \rangle =
\frac{1}{2} \frac{\Lambda}{\gamma^2} \left\langle \frac{\partial^2 A}{\partial x^2} \right\rangle
+ \frac{1}{\gamma} \left\langle F(x) \frac{\partial A}{\partial x} \right\rangle
\]

\textbf{אבולוציה של ווריאנס:}
\[
\frac{d}{dt} \langle A^2(x) \rangle =
\frac{1}{2} \frac{\Lambda}{\gamma^2} \left\langle \frac{\partial^2 A^2}{\partial x^2} \right\rangle
+ \frac{1}{\gamma} \left\langle F(x) \frac{\partial A^2}{\partial x} \right\rangle
\]
}{0.46}
\end{mytikzbox}
\begin{mytikzbox}{פתרון סטציונרי של משוואת פוקר-פלאנק}{
הפתרונות הסטציונריים מקיימים:
\[
0=\frac12\frac{\Lambda}{\gamma^{2}}\frac{d^{2}}{d x^{2}}P\left(x\right)-\frac1{\gamma}\frac d{d x}\left[P\left(x\right)F\left(x\right)\right]
\]
נסדר מחדש:
\[
\frac{d}{d x}\left[P\left(x\right)F\left(x\right)\right]=\frac{1}{2}\frac{\Lambda}{\gamma}\frac{d^{2}}{d x^{2}}P\left(x\right)
\]
ניקח אינטגרל:
\[
P\left(x\right)F\left(x\right)=\frac{1}{2}\frac{\Lambda}{\gamma}\frac{d}{d x}P\left(x\right)+C
\]
כיוון ש-$P(x)\xrightarrow{x\to \pm \infty}0$ ו-$P'(x)\xrightarrow{x\to \pm \infty}0$ נקבל כי $C=0$:
\[
P\left(x\right)F\left(x\right)=\frac{1}{2}\frac{\Lambda}{\gamma}\frac{d P}{d x}
\]
זו משוואה ספרבילית:
\[
\int F\left(x\right)d x=\frac{1}{2}\frac{\Lambda}{\gamma}\int\frac{1}{P}d P=\frac{1}{2}\frac{\Lambda}{\gamma}\ln\left(Z P\right)
\]
ולכן:
\[
P\left(x\right)=\frac{1}{Z}e^{\frac{2\gamma}{\Lambda}\int F(x)\,dx}
\]
אם $F(x) = -\frac{dU}{dx}$ (כלומר כוח משמר):
\[
P\left(x\right)=\frac{1}{Z}e^{-\frac{2\gamma}{\Lambda}U(x)}
\]
שזה התפלגות בולצמן תחת הזיהו $\frac{\Lambda}{2\gamma}=k_{B}T$.
}
{0.46}
\end{mytikzbox}
\begin{mytikzbox}{חלקיק ב-\(d\) מימדים בפוטנציאל הרמוני: פוקר-פלאנק}{
נתונה המשוואת לנג'וון מרוסן יתר עם פוטנציאל הרמוני $U=\frac{1}{2}k|\vec{x}|^2$:
\[
\gamma \frac{d\vec{x}}{dt} = -k\vec{x} + \vec{\eta}(t)
\]
כאשר $\langle \eta_i(t) \eta_j(t') \rangle = \Lambda \delta_{ij} \delta(t-t')$. נתון כי החלקיק מתחיל במיקום $\vec{x}_0$ בזמן $t=0$.

\textbf{1. נכתוב את פוקר-פלאנק:} 
$$\frac{\partial}{\partial t}P\left(\vec{x},t\right)=\sum_{i=1}^{d}\left(\frac{1}{2}\frac{\Lambda}{\gamma^{2}}\frac{\partial^{2}}{\partial x_{i}^{2}}P\left(\vec{x},t\right)+\frac{k}{\gamma}\frac{\partial}{\partial x_{i}}\left[x_{i}P\left(\vec{x},t\right)\right]\right)$$
\[\Rightarrow\;\;
\frac{\partial}{\partial t} P(\vec{x}, t) = \frac{1}{2} \frac{\Lambda}{\gamma^2} \nabla^2 P + \frac{k}{\gamma} \nabla \cdot (\vec{x} P)
\]

\textbf{2. נכתוב את הפונקציה יוצרת מומנטים:} ראינו כי אם נכפיל בפונקציה מרחבית $A(\vec{x})$ ונבצע אינטרקציה על המרחב:
$$\frac{\partial}{\partial t}\left\langle A\left(\vec{x}\right)\right\rangle=\frac{1}{2}\frac{\Lambda}{\gamma^{2}}\left\langle\nabla^{2}A\right\rangle-\frac{k}{\gamma}\left\langle\vec{x}\cdot\vec{\nabla}A\right\rangle$$
נשתמש בפונקציה $A(\vec{x})=e^{\vec{z} \cdot \vec{x}}=M$:
\[
M(\vec{z}, t) = \langle e^{\vec{z} \cdot \vec{x}} \rangle = \int d^d x\, e^{\vec{z} \cdot \vec{x}} P(\vec{x}, t)
\]
מקיימת:
\[
\frac{\partial M}{\partial t} = -\frac{k}{\gamma} \vec{z} \cdot \nabla_{\vec{z}} M + \frac{1}{2} \frac{\Lambda}{\gamma^2} |\vec{z}|^2 M
\]
עם תנאי התחלה $M(\vec{z},0) = e^{\vec{z}\cdot\vec{x}_0}$.

\textbf{3. הניחוש:} ראינו שניחוש טוב הוא גאוסיאן. נניח $M(\vec{z}, t) = \exp\left( \vec{A}(t)\cdot\vec{z} + B(t)|\vec{z}|^2 \right)$.
נציב ונקבל:
\[
\vec{A}(t) = \vec{x}_0 e^{-kt/\gamma}
\qquad
B(t) = \frac{\Lambda}{2k\gamma}(1 - e^{-2kt/\gamma})
\]
ולכן:
\[
M(\vec{z}, t) = \exp\left[ \vec{z} \cdot \vec{x}_0 e^{-kt/\gamma} + \frac{\Lambda}{2k\gamma} (1 - e^{-2kt/\gamma}) |\vec{z}|^2 \right]
\]
כלומר, $M$ היא פונקציה יוצרת של גאוסיאן רב-ממדי.

\textbf{ממוצע ווריאנס:}
\[
\langle \vec{x}(t) \rangle = \vec{x}_0 e^{-kt/\gamma}
\qquad
\mathrm{Var}(x_i) = \frac{\Lambda}{2k\gamma} (1 - e^{-2kt/\gamma})
\]

\textbf{התפלגות ההסתברות:}
\[
P(\vec{x}, t) = \frac{1}{[2\pi \sigma^2(t)]^{d/2}} \exp\left( -\frac{|\vec{x} - \vec{x}_0 e^{-kt/\gamma}|^2}{2\sigma^2(t)} \right)
\]
\[
\sigma^2(t) = \frac{\Lambda}{2k\gamma} (1 - e^{-2kt/\gamma})
\]

\textbf{גבול $t\to\infty$:} מקבלים התפלגות בולצמן:
\[
P_{\mathrm{eq}}(\vec{x}) \propto \exp\left( -\frac{k|\vec{x}|^2}{\Lambda/(2\gamma)} \right)
\]
}{0.46}
\end{mytikzbox}
