\begin{mytikzbox}{Thermodynamics and potentials}{0.475}
\textbf{Thermodynamic potentials.}
Legendre: $p=\partial f/\partial x$,\ 
$f^*(p)=px(p)-f(x(p))$.
\begin{center}

\begin{tikzpicture}[
      main/.style={anchor=center},

\end{center}
  \begin{align*}
  & E = TS - PV + \mu N & dE &= TdS - PdV + \mu dN\\
  & H = E + PV = TS + \mu N & dH &= TdS + VdP + \mu dN\\
  & F = E - TS = -PV + \mu N & dF &= -SdT - PdV + \mu dN\\
  & G = E - TS + PV = \mu N & dG &= -SdT + VdP + \mu dN\\
  & \Omega = F - \mu N = -PV & d\Omega &= -SdT - PdV - Nd\mu
  \end{align*}\\[-2pt]
\textit{$E=TS-PV+\mu N$ is the Euler relation for an extensive homogeneous system.}

\textbf{Processes:} isothermal $\dd T=0$; isochoric $\dd V=0$;
isobaric $\dd P=0$; adiabatic $\delta Q=0$; isentropic $\dd S=0$.
For a reversible path $\dd S=\delta Q_{\rm rev}/T$; generally
$\dd S\geq\delta Q/T$. A reversible adiabatic process is isentropic.
\end{mytikzbox}

\begin{mytikzbox}{Thermodynamic quantities}{0.475}
\[
\beta=\frac1{\kb T},\qquad
\frac{\partial\beta}{\partial T}=-\frac1{T^2\kb}=-\beta^2\kb,\qquad
C=\frac{\partial U}{\partial T}.
\]

J_i\equiv\frac{\partial E(S,x_i)}{\partial x_i},\qquad

\]

E(S,x_i)=T(S,x_i)S+\sum_i J_i x_i

\begin{mytikzbox}{Entropy and maximum entropy}{0.475}
\textbf{Boltzmann:} for an isolated macrostate with multiplicity $\Omega$,
\[
S_{\rm B}=\kb\ln\Omega,
\qquad \frac1T=\left(\frac{\partial S}{\partial E}\right)_{V,N}.
\]
\textbf{Gibbs--Shannon:}
\[
S[p]=-\kb\sum_i p_i\ln p_i,
\qquad \sum_i p_i=1.
\]
Maximizing $S[p]$ subject to $\avg{A_a}=\sum_i p_i A_{a,i}$ gives
\[
p_i=\frac1Z\exp\!\left(-\sum_a\lambda_a A_{a,i}\right),
\qquad Z=\sum_i\exp\!\left(-\sum_a\lambda_a A_{a,i}\right).
\]
Canonical constraint $\avg E$: $\lambda_E=\beta=1/(\kb T)$ and
$p_i=Z^{-1}\e^{-\beta E_i}$. Equal probabilities apply in the
\emph{isolated total system}; tracing out a heat bath produces the
Boltzmann factor $\Omega_B(E_{\rm tot}-E_i)\propto\e^{-\beta E_i}$.
\end{mytikzbox}

\begin{mytikzbox}{Transforms and ensemble choice}{0.475}
\textbf{Legendre transform:} if $p=f'(x)$,
$f^*(p)=px(p)-f(x(p))$.
It exchanges a controlled extensive variable with its intensive conjugate.

\textbf{Laplace transform:}
\[
\mathcal L\{g\}(s)=\int_0^\infty \e^{-sx}g(x)\dd x.
\]
Canonical $Z(\beta)$ is the Laplace transform of the density of states:
$Z(\beta)=\int\rho(E)\e^{-\beta E}\dd E$.

\textbf{$z$-transform / generating function:}
\[
\widetilde a(z)=\sum_{N\ge0}a_Nz^N,
\qquad Z_\Omega(z)=\sum_{N\ge0}z^N Z_F,
\qquad z=\e^{\beta\mu}.
\]
Mostly used in the context of cluster expansion.
\[
\begin{array}{c|c|c}
\text{fixed variables}&\text{ensemble}&\text{potential}\\ \hline
E,V,N&\text{microcanonical}&E\ \text{or }S\\
T,V,N&\text{canonical}&F\\
T,P,N&\text{isothermal--isobaric}&G\\
T,V,\mu&\text{grand canonical}&\Omega
\end{array}
\]
\end{mytikzbox}

\begin{mytikzbox}{Partition functions and derivatives}{0.475}
With weights \(z^i\) and \(Z=\sum_i z^i\),
\[
\avg A \equiv \sum A_i p_i = Z^{-1}\sum_i A_i\,z^i.
\]

\textbf{Canonical ensemble $(T,V,N)$}
\[
Z_F=\sum_i\e^{-\beta E_i},\quad F=-\kb T\ln Z_F,
\quad U=-\partial_\beta\ln Z_F,
\]
\[
S=\kb(\ln Z_F+\beta U),\quad
P=\kb T\,\partial_V\ln Z_F,
\]
\[
\Var(E)=\partial_\beta^2\ln Z_F=\kb T^2C_V.
\]
\textbf{Cumulants / generating functions.}
\[
M_X(t)\equiv\avg{\e^{tX}},
\quad
K_X(t)\equiv\ln M_X(t),
\]
\[
\avg{X^n}=M_X^{(n)}(0),
\quad
\kappa_n(X)=K_X^{(n)}(0).
\]
Canonical energy: $M_E(t)=Z_F(\beta-t)/Z_F(\beta)$, hence
\[
\kappa_n(E)=(-1)^n\partial_\beta^n\ln Z_F,
\]
so $\avg E=-\partial_\beta\ln Z_F$ and $\Var(E)=\partial_\beta^2\ln Z_F$.
Grand canonical:
\[
\kappa_n(N)=\partial_{(\beta\mu)}^n\ln Z_\Omega=(z\partial_z)^n\ln Z_\Omega.
\]
For nonnegative integer $N$, the PGF is $G_N(z)=\avg{z^N}=\sum_n P_n z^n$,
with $M_N(t)=\avg{\e^{tN}}=G_N(\e^t)$,
$P_n=G_N^{(n)}(0)/n!$,
$G_N'(1)=\avg N$,
$G_N''(1)=\avg{N(N-1)}$.

Classical phase space:
\[
Z_F=\frac1{N!h^{3N}}\int\prod_{i=1}^N\dd^3p_i\dd^3q_i\,
\e^{-\beta H_N}.
\]

% \quad G=-\kb T\ln Z_G,
% \]

\textbf{Gibbs ensemble $(T,P,N)$}
\[
Z_G(\beta,P,N)
=
\int_0^\infty \dd V\,\e^{-\beta PV}Z_F(\beta,V),
\qquad
G=-\kb T\ln Z_G.
\]
Here the effective energy is the enthalpy (Laplace transform of $Z_F$ in $V$)
\[
H=E+PV
\]
\[
\Rightarrow\avg H=-\partial_\beta\ln Z_G,
\qquad
\avg V=-\frac{1}{\beta}\partial_P\ln Z_G,
\]
\[
S=\kb\ln Z_G+\kb\beta\avg H.
\]

For spins in a magnetic field,
\[
B\leftrightarrow -P,
\qquad
M=\sum_i s_i\leftrightarrow V,
\qquad
-BM\leftrightarrow PV.
\]

\textbf{Grand-canonical ensemble $(T,V,\mu)$}
\[
Z_\Omega(\beta,V,\mu)
=
\sum_{N=0}^{\infty} e^{\beta \mu N} Z_F(\beta, V, N)=\sum_{N}z^{N}Z_F(\beta,V,N)
\]
The grand potential is
\[
\Omega = -\frac{1}{\beta} \ln Z_\Omega = -PV
\]
The mean particle number and its fluctuations are
\[
\avg N
=
\frac{1}{\beta}\partial_\mu\ln Z_\Omega
=
z\partial_z\ln Z_\Omega,
\]
\[
\Var(N)
=
\frac{1}{\beta^2}\partial_\mu^2\ln Z_\Omega.
\]
The entropy is
\[
S
=
\kb\ln Z_\Omega
+
\kb\beta\left(\avg E-\mu\avg N\right).
\]
\end{mytikzbox}

\begin{mytikzbox}{Ideal gas and equipartition}{0.475}
Thermal wavelength $\lambda_T=h/\sqrt{2\pi m\kb T}$:
\[
Z_F=\frac1{N!}\left(\frac{V}{\lambda_T^3}\right)^N,
\quad PV=N\kb T,
\quad U=\frac32N\kb T.
\]
Using Stirling, $\ln N!\simeq N\ln N-N$:
\[
F=-N\kb T\left[\ln\left(\frac{V}{N\lambda_T^3}\right)+1\right],
\quad
S=N\kb\left[\ln\left(\frac{V}{N\lambda_T^3}\right)+\frac52\right].
\]
Every independent quadratic term in $H$ contributes $\frac12\kb T$ to
$U$ and $\frac12\kb$ to $C_V$.
More generally (boundary terms vanishing),
\[
\avg{x_i\,\partial_{x_j}H}=\kb T\,\delta_{ij},
\]
so $H\supset ax^2$ gives $a\avg{x^2}=\tfrac12\kb T$
(or $H\supset\tfrac12 a x^2$ gives $\tfrac12 a\avg{x^2}=\tfrac12\kb T$).
\end{mytikzbox}

\begin{mytikzbox}{Laplace / saddle-point method}{0.475}
For $N\gg1$, $\e^{-Nf(x)}$ is dominated by the minimum $x_*$ of $f$:
\[
I(N)=\int g(x)\e^{-Nf(x)}\dd x
\simeq g(x_*)\e^{-Nf(x_*)}
\sqrt{\frac{2\pi}{Nf''(x_*)}}.
\]
First correction ($f'(x_*)=0$, write $f_*=f(x_*)$, etc.):
\[
\begin{aligned}
I(N)&\sim
\e^{-Nf_*}\sqrt{\frac{2\pi}{Nf''_*}}
\bigl[g_*+\tfrac1N\delta_1+\cdots\bigr],\\
\delta_1&=
\frac{g''_*}{2f''_*}
-\frac{g'_*f'''_*}{2(f''_*)^2}
-\frac{g_*f''''_*}{8(f''_*)^2}
+\frac{5g_*(f'''_*)^2}{24(f''_*)^3}.
\end{aligned}
\]
In $d$ dimensions, with Hessian $H_{ij}=\partial_i\partial_jf(x_*)$,
\[
I(N)\simeq g(x_*)\e^{-Nf(x_*)}
\left(\frac{2\pi}{N}\right)^{d/2}(\det H)^{-1/2}.
\]
Width $\Delta x\sim[Nf''(x_*)]^{-1/2}$;
\textbf{small parameter:} $1/N$ (or $\beta$ at low $T$).
Boundary / degenerate minima need a modified expansion.

\begin{mytikzbox}{Transfer matrix: periodic 1D Ising model}{0.475}
\[
H=-J\sum_{i=1}^Ns_is_{i+1}-h\sum_{i=1}^Ns_i,
\qquad s_i=\pm1,\text{ }s_{N+1}=s_1.
\]
Split the field equally between adjacent bonds:
\[
T_{ss'}=\exp\!\left[\beta Jss'+\frac{\beta h}{2}(s+s')\right],
\quad
T=\begin{pmatrix}
\e^{\beta J+\beta h}&\e^{-\beta J}\\
\e^{-\beta J}&\e^{\beta J-\beta h}
\end{pmatrix}.
\]
\[
Z_N=\Tr T^N=\lambda_+^N+\lambda_-^N,
\]
\[
\lambda_\pm=\e^{\beta J}\cosh(\beta h)
\pm\sqrt{\e^{2\beta J}\sinh^2(\beta h)+\e^{-2\beta J}}.
\]
For $N\to\infty$, $f=-\kb T\ln\lambda_+$ and
$m=\partial_{\beta h}\ln\lambda_+$. Insert the diagonal spin operator
$S=\operatorname{diag}(1,-1)$:
\[
\avg{s_0s_r}=\frac{\Tr(S T^r S T^{N-r})}{\Tr T^N}.
\]
If $T=\sum_a\lambda_a|a\rangle\langle a|$ with $|\lambda_0|>|\lambda_1|\ge\cdots$,
then for $N\to\infty$
\[
\avg{A_0B_r}
=\sum_a\left(\frac{\lambda_a}{\lambda_0}\right)^r
\langle0|A|a\rangle\langle a|B|0\rangle,
\]
so $\avg{A_0B_r}_c\sim(\lambda_1/\lambda_0)^r$ and
$\xi^{-1}=-\ln|\lambda_1/\lambda_0|$.
\textbf{At $h=0$:} with $q\equiv\lambda_-/\lambda_+$,
\[
\avg{s_0s_r}=\frac{q^r+q^{N-r}}{1+q^N}.
\]
As $N\to\infty$:
\[
\avg{s_0s_r}=q^r=(\tanh\beta J)^r=\e^{-r/\xi},
\qquad \xi^{-1}=-\ln|q|=-\ln(\tanh\beta J).
\]
A phase transition would require correlations to persist over arbitrarily large distances, i.e. $\xi\to\infty$. In the 1D Ising model, $|\lambda_1|<\lambda_0$ for every $T>0$, so $\xi$ remains finite and there is no finite-temperature transition. Generally, the subleading eigenvalue sets the slowest-decaying correction and therefore controls long-range correlations.
\end{mytikzbox}

\begin{mytikzbox}{Ising mean-field theory}{0.475}
Consider a system with nearest-neighbor interactions, \[ H_{\mathrm{int}}=-J\sum_{\langle i,j\rangle}s_i s_j. \] Define the fluctuation around the mean magnetization by \[ \delta s_i=s_i-m. \] Let \(d\) be the spatial dimension and \(z=2d\) the coordination number, i.e. the number of nearest neighbors of each site. 

\textbf{Partition function:} \[ Z_G=\operatorname{tr}\exp\left\{ \beta\left( J\sum_{\langle i,j\rangle}s_i s_j +B\sum_i s_i \right) \right\}. \] Using \(s_i=m+\delta s_i\), using $[\cdots] = \delta s_i\delta s_j +m(\delta s_i+\delta s_j)\, +m^2$\[ Z_G= \operatorname{tr}\exp\left\{ \beta J\sum_{\langle i,j\rangle} \left[ \cdots \right] +\beta B\sum_i\delta s_i +\beta BNm \right\}. \] 

\textbf{Mean-field assumption:} \[ \delta s_i\delta s_j\approx0 \implies s_i s_j\approx m(s_i+s_j)-m^2. \] 

\textbf{Approximate partition function:} Since \[ \sum_{\langle i,j\rangle}1=\frac{Nz}{2}, \] we obtain \[ Z_G^{\mathsf{MF}} = \exp\left(-\frac{\beta JNz m^2}{2}\right) \left[ 2\cosh\left(\beta B_{\mathsf{MF}}\right) \right]^N, \] where \[ B_{\mathsf{MF}}=B+Jzm. \] The self-consistency equation for the magnetization is \[ m= \frac{1}{\beta N} \frac{\partial\ln Z_G}{\partial B} = \tanh\left[\beta(B+Jzm)\right]. \]
The mean-field free energy per spin (extrema give that equation) is
\[
f_{\rm MF}(m)=\frac{Jz}{2}m^2
-\frac1\beta\ln\!\bigl[2\cosh\bigl(\beta(B+Jzm)\bigr)\bigr]
\]
(up to an additive constant). Then
$\partial_m f_{\rm MF}=0\Leftrightarrow m=\tanh[\beta(B+Jzm)]$,
and $\partial_m^2 f_{\rm MF}>0$ selects stable solutions.

\textbf{Power-laws}
At $h=0$, $T_c=zJ/\kb$. Near $T_c$:
\[
m\sim(-t)^{1/2},\quad \chi\sim |t|^{-1},\quad
m(T_c,h)\sim h^{1/3},\qquad t=(T-T_c)/T_c.
\]
\end{mytikzbox}

\begin{mytikzbox}{Hubbard--Stratonovich transformation}{0.475}
For $a>0$,
\[
\e^{\frac a2x^2}=\frac1{\sqrt{2\pi a}}
\int_{-\infty}^{\infty}\dd\phi\,
\exp\!\left[-\frac{\phi^2}{2a}+\phi x\right].
\]
For a positive matrix $K$,
\[
\e^{\frac12\bm s^T K\bm s}
=\frac1{\sqrt{\det(2\pi K)}}\int\dd^n\phi\,
\e^{-\frac12\bm\phi^TK^{-1}\bm\phi+\bm\phi^T\bm s}.
\]
The Hubbard--Stratonovich transformation trades a direct quadratic interaction between the microscopic variables for a linear interaction with a new auxiliary field. The new field $\phi$ is continuous and is integrated over all real values.

After the microscopic variables are summed or integrated out, $\phi$ acquires an effective action and describes fluctuations of the order parameter. Approximating the remaining integral by its dominant saddle point fixes $\phi$ to its most probable value and reproduces the mean-field approximation.
\end{mytikzbox}

\begin{mytikzbox}{Landau and Ginzburg--Landau theory}{0.475}
Landau theory uses a coarse-grained order parameter $\varphi$ (e.g.\ magnetization).
(After HS, integrating out the microscopic degrees of freedom turns the auxiliary field
$\phi$ into this Landau order parameter $\varphi$.)
Above $T_c$, symmetry favors $\varphi=0$; below $T_c$ the system may choose $\varphi\neq0$.
Expand the free-energy density about $\varphi=0$:
\[
f(T,\varphi)=f_0(T)+\frac12 a(T)\varphi^2+\frac14 u\,\varphi^4-h\varphi+\cdots,
\]
with $u>0$ and $a(T)\approx a_0(T-T_c)=a_0 T_c\,t$, $t=(T-T_c)/T_c$, $a_0>0$.
Equivalently,
\[
f(\varphi)=f_0+\frac{a_0t}{2}\varphi^2+\frac u4\varphi^4-h\varphi+\cdots.
\]
\textbf{Minimize:} at $h=0$,
\[
\frac{\partial f}{\partial\varphi}=0
\implies a(T)\varphi+u\varphi^3=0.
\]
Solutions: $\varphi=0$ always, and
$\varphi=\pm\sqrt{-a(T)/u}$ when $a(T)<0$ (i.e.\ $T<T_c$).
Near $T_c$,
\[
\varphi(T)=\pm\sqrt{\frac{a_0(T_c-T)}{u}}
\sim(-t)^{1/2},
\qquad\Rightarrow\quad \beta_{\rm MF}=\tfrac12.
\]
If $u>0$ the order parameter grows continuously from zero
(\textbf{second-order} / continuous transition).
If the quartic coefficient is negative, a stabilizing $+v\varphi^6$ term is needed and the
transition is first order; $u=0$ is the tricritical point.

Ginzburg--Landau allows $\varphi$ to vary in space:
\[
\mathcal F[\varphi]=\int\dd^dx\left[
\frac c2(\grad\varphi)^2+\frac r2\varphi^2+\frac u4\varphi^4-h\varphi\right],
\quad Z=\int\mathcal D\varphi\,\e^{-\beta\mathcal F[\varphi]}.
\]
The gradient penalizes rapid spatial changes; the remaining terms set the locally
preferred $\varphi$. Minimizing $\mathcal F$ gives
\[
-c\nabla^2\varphi+r\varphi+u\varphi^3=h.
\]
Valid near a continuous transition for slow variations on scales $\gg$ lattice spacing
(long-wavelength order-parameter fluctuations).
\end{mytikzbox}

\begin{mytikzbox}{Critical exponents, scaling and universality}{0.475} Near a continuous transition, physical quantities follow power laws: \[ C\sim|t|^{-\alpha},\quad m\sim(-t)^\beta, \quad\chi\sim|t|^{-\gamma},\quad m(t=0,h)\sim h^{1/\delta}, \] \[ \xi\sim|t|^{-\nu},\qquad G(r,t=0)\sim r^{-(d-2+\eta)}. \] These powers define the critical exponents. They appear because, as \(t\to0\), the correlation length diverges and the system becomes approximately scale invariant. Widom's scaling hypothesis assumes that the singular part of the free-energy density has the form \[ f_s(t,h)=|t|^{2-\alpha}\mathcal F_\pm \!\left(h/|t|^{\beta\delta}\right). \] Thus \(t\) and \(h\) do not enter independently, but only through a scale-free combination. Taking derivatives of \(f_s\), for example \[ m=-\partial_h f_s,\qquad \chi=\partial_h m, \] produces the power laws and relations between their exponents: \[ m=|t|^\beta\mathcal M_\pm \!\left(h/|t|^{\beta\delta}\right), \] \[ \alpha+2\beta+\gamma=2, \qquad \gamma=\beta(\delta-1), \qquad \gamma=\nu(2-\eta), \qquad 2-\alpha=d\nu\quad. \] Mean-field values for scalar \(\phi^4\) theory are (see 3 boxes above) \[ (\alpha,\beta,\gamma,\delta,\nu,\eta) =(0,\tfrac12,1,3,\tfrac12,0). \] Universality means that the exponents depend mainly on spatial dimension, symmetry, number of order-parameter components and interaction range, rather than on microscopic details. \end{mytikzbox}

\begin{mytikzbox}{Fourier-space intuition}{0.475}
For a lattice field $s_i$ at positions $\bm r_i$,
\[
s_i=\frac1{\sqrt N}\sum_{\bm k}s_{\bm k}\e^{i\bm k\cdot\bm r_i},
\qquad
s_{\bm k}=\frac1{\sqrt N}\sum_i s_i\e^{-i\bm k\cdot\bm r_i}.
\]
Translation-invariant quadratic forms diagonalize:
\[
\frac12\sum_{ij}s_iK_{i-j}s_j
=\frac12\sum_{\bm k}K(\bm k)|s_{\bm k}|^2.
\]
Near a transition the soft modes have $ka\ll1$, so
$K(\bm k)\simeq r+c k^2+\cdots$; inverse Fourier transformation gives the
Ginzburg--Landau gradient term.
\end{mytikzbox}

\begin{mytikzbox}{Gaussian functional integrals and Green functions}{0.475}
For a positive operator $K$ and dimensionless source $J$,
\[
Z[J]=\int\mathcal D\phi\,
\exp\!\left[-\frac12\phi K\phi+J\phi\right]
=Z[0]\exp\!\left(\frac12J K^{-1}J\right).
\]
\[
\avg{\phi(x)}_J=\frac{\delta\ln Z}{\delta J(x)},
\]
\[
G_c(x,y)=\langle \phi(x) \phi(y) \rangle_J = \frac{\delta^2\ln Z}{\delta J(x)\delta J(y)}=K^{-1}(x,y).
\]
For the Gaussian Ginzburg--Landau theory,
\[
K(\bm k)=r+ck^2,
\qquad \widetilde G(\bm k)=\frac1{r+ck^2},
\qquad \xi=\sqrt{c/r}.
\]
At large $R=|\bm x|$,
\[
G(R)\sim\frac{\e^{-R/\xi}}{R^{(d-1)/2}}
\quad (R\gg\xi),
\qquad
G(R)\sim R^{-(d-2)}\quad(r=0,\ d>2).
\]
If the physical field couples as $-h\int\phi$, then
\[
\chi=\frac{\partial\avg\phi}{\partial h}
=\beta\int\dd^dx\,G_c(x,0)=\beta\widetilde G(0).
\]
When $J=\beta h$ is used as the source, the same statement is
$\partial\avg\phi/\partial J=\widetilde G(0)$.
\end{mytikzbox}

\begin{mytikzbox}{Goldstone modes}{0.475} An $O(n)$ field is an $n$-component real field, $\vec{\phi}(\vec{x}) = (\phi_1(\vec{x}),...,\phi_n(\vec{x}))$, whose free energy is invariant under rotations $R\in O(n)$. 
  
For an \(O(n)\) field, \[ \mathcal F=\int\dd^dx\left[ \frac c2(\grad\bm\phi)^2+\frac r2\bm\phi^2+ \frac u4(\bm\phi^2)^2\right],\qquad r<0. \] The minima form a continuous sphere of equivalent ordered states. Spontaneous symmetry breaking means that the system chooses one direction, for example \[ \avg{\bm\phi}=v\hat e_1,\qquad v^2=-r/u, \] and we decompose \[ \bm\phi=(v+\sigma,\pi_1,\ldots,\pi_{n-1}). \] The longitudinal fluctuation \(\sigma\) changes the magnitude of the order parameter and moves away from the minimum, so it costs local energy. The transverse fluctuations \(\pi_a\) rotate the order parameter between equivalent minima, so a uniform rotation costs no energy: \[ m_L^2=2|r|\quad(\sigma\text{, longitudinal}), \qquad m_T^2=0\quad(\pi_a\text{, transverse}). \] Thus spontaneous breaking \(G\to H\) creates \(\dim G-\dim H\) massless Goldstone modes: one for each independent direction along the degenerate vacuum manifold. For \(O(n)\to O(n-1)\), there are \(n-1\) such modes, with \[ G_T(k)\sim\frac1{ck^2}. \] Only spatial variations of these rotations cost gradient energy, which vanishes for sufficiently long wavelengths. 

Mermin-Wagner theorem: In local-interaction systems, there is no finite-temperature continuous symmetry breaking in $d<2$ dimensions.
\end{mytikzbox}

\begin{mytikzbox}{Upper/lower critical dimensions and Ginzburg criterion}{0.475} The lower critical dimension asks: in how few dimensions can an ordered phase survive fluctuations? For a system with a broken continuous symmetry, the massless Goldstone modes have \[ G_T(k)\sim\frac{1}{k^2}, \qquad \avg{\pi^2}\sim\int_{k_{\min}}\frac{\dd^dk}{k^2}. \] For \(d\le2\), the long-wavelength contribution \(k\to0\) diverges. Large-scale rotations then destroy the chosen direction of the order parameter, so there is no finite-\(T\) continuous symmetry breaking in short-range systems (Mermin--Wagner). Hence \(d_\ell=2\) for \(O(n\ge2)\). For the discrete Ising symmetry there are no Goldstone modes, and \(d_\ell=1\). 
  
The upper critical dimension asks: above which dimension are fluctuations weak enough that mean-field theory becomes correct? For \(\phi^4\), \[ g(\xi)\sim \frac{u\kb T}{c^2}\,\xi^{4-d} \] measures the strength of fluctuations on the correlation-length scale. Mean field is self-consistent when \(g(\xi)\ll1\) (Ginzburg criterion). Since \(\xi\to\infty\) near the transition, fluctuations grow for \(d<4\), are marginal at \(d=4\), and become irrelevant for \(d>4\). Therefore \(d_c=4\): below it the exponents are non-mean-field, at it logarithmic corrections appear, and above it mean-field theory is asymptotically correct. Hyperscaling generally fails above \(d_c\). \end{mytikzbox}

\begin{mytikzbox}{Mayer expansion and linked clusters}{0.475}
For pair interactions $U=\sum_{i<j}u(r_{ij})$ define
\[
f_{ij}=\e^{-\beta u(r_{ij})}-1,
\qquad \e^{-\beta U}=\prod_{i<j}(1+f_{ij}).
\]
Each term corresponds to a graph: particles are vertices and $f_{ij}$ are edges.
The configurational integral is a sum over all graphs. Every graph factorizes
into connected components; exponentiating the connected-cluster sum gives
\[
\ln Z_\Omega=V\sum_{\ell\ge1}b_\ell z^\ell,
\quad
\beta P=\sum_{\ell\ge1}b_\ell z^\ell,
\quad
n=\sum_{\ell\ge1}\ell b_\ell z^\ell,
\]
where this convention absorbs the powers of $\lambda_T$ into $z,b_\ell$.
The cluster coefficients keep only \emph{connected} graphs $\mathcal C_\ell$:
\[
b_\ell=\frac1{\ell!V}\int_{V^\ell}
\Biggl(\sum_{G\in\mathcal C_\ell}\prod_{(i,j)\in G}f_{ij}\Biggr)
\prod_{i=1}^\ell\dd^dr_i.
\]
The logarithm drops disconnected graphs because they factorize into products
already generated by the exponential.

\textbf{Example}
\(\ell=3\): The connected graphs are the three open chains and the triangle: \[ \sum_{G\in\mathcal C_3}\prod f_{ij} = f_{12}f_{13}+f_{12}f_{23}+f_{13}f_{23} +f_{12}f_{13}f_{23}. \] Hence \[ b_3=\frac1{3!V}\int_{V^3} \bigl[f_{12}f_{13}+f_{12}f_{23}+f_{13}f_{23} +f_{12}f_{13}f_{23}\bigr] \prod_{i=1}^3\dd^dr_i. \] The open chains are reducible: removing their central particle disconnects the graph. The triangle remains connected after removing any one particle and is therefore irreducible. Connected graphs determine \(\ln Z_\Omega\), while the virial coefficients contain only irreducible graphs; consequently, only the triangle contributes directly to \(B_3\). For a partition with \(n_k\) clusters of size \(k\), \[ \sum_kkn_k=N, \qquad \Omega_N(\{n_k\})= \frac{N!}{\prod_k(k!)^{n_k}n_k!}. \] 
If unlabeled particles, must take into account topology - multiply $b_\ell$ by $\ell! / | \text{Aut}(G)|$.
\end{mytikzbox}

\begin{mytikzbox}{Virial expansion and basic coefficients}{0.475}
The grand-canonical calculation naturally gives pressure and density as functions of the fugacity \(z\). Eliminating \(z\) between \(P(z)\) and \(n(z)\) gives the virial expansion, \[ \beta P=n+B_2n^2+B_3n^3+\cdots. \] The first term is the ideal-gas law; the higher powers describe corrections due to interactions between pairs, triples, and larger groups of particles. If \[ \beta P=z+b_2z^2+b_3z^3+\cdots, \] then \[ B_2=-b_2, \qquad B_3=4b_2^2-2b_3. \] For a translation-invariant pair potential, \[ B_2=-\frac12\int\dd^dr\,f(r), \] \[ B_3=-\frac13\int\dd^dr_{12}\dd^dr_{13}\, f_{12}f_{13}f_{23}. \] Thus \(B_2\) measures the effect of one interacting pair, while \(B_3\) contains the genuinely three-particle contribution in which all three particles are linked. The virial expansion is useful experimentally because density is measurable, whereas fugacity is not. 

For hard spheres of diameter \(\sigma\) in three dimensions, \[ B_2=\frac{2\pi}{3}\sigma^3, \qquad B_3=\frac{5\pi^2}{18}\sigma^6=\frac58B_2^2. \] Here \(B_2\) is half the volume excluded to the center of a second sphere. More generally, for a \(d\)-dimensional hard core, \[ B_2=\tfrac12V_d(\sigma), \qquad V_d(R)=\frac{\pi^{d/2}R^d}{\Gamma(d/2+1)}. \]
\end{mytikzbox}

\begin{mytikzbox}{Markov processes and Chapman--Kolmogorov}{0.475}
A process is Markovian when the future conditioned on the present is independent
of the past:
\[
P(x_n,t_n|x_{n-1},t_{n-1};\ldots;x_0,t_0)
=P(x_n,t_n|x_{n-1},t_{n-1}).
\]
Transition kernels obey Chapman--Kolmogorov:
\[
P(x_2,t_2|x_0,t_0)=\int\dd x_1\,
P(x_2,t_2|x_1,t_1)P(x_1,t_1|x_0,t_0).
\]
For a homogeneous discrete chain, $\bm P_{n+1}=T\bm P_n$,
$\sum_iT_{ij}=1$, where $T_{ij}$ is the transition probability
\textbf{from state $j$ to state $i$} ($j\to i$).
(Column-stochastic: columns sum to 1.)

A stationary distribution is an eigenvector with eigenvalue
$1$; modes with $|\lambda|<1$ decay. Brownian motion is approximately Markovian
when molecular collision times are much shorter than the observed time scale.
\end{mytikzbox}

\begin{mytikzbox}{Wiener process and Gaussian white noise}{0.475}
A Wiener process $W(t)$ has $W(0)=0$, continuous paths, independent stationary
increments, and
\[
W(t+\Delta t)-W(t)\sim\mathcal N(0,\Delta t),
\quad \avg{W(t)}=0,
\]\[
\avg{W(t)W(s)}=\min(t,s).
\]
Gaussian white noise is the distributional derivative $\eta(t)=\dot W(t)$:
\[
\avg{\eta(t)}=0,
\qquad \avg{\eta(t)\eta(t')}=\delta(t-t').
\]
For strength $\Lambda$, write $\xi(t)=\sqrt\Lambda\,\eta(t)$, so
$\avg{\xi(t)\xi(t')}=\Lambda\delta(t-t')$. On a grid,
\[
\avg{\xi_i\xi_j}=\frac{\Lambda}{\Delta t}\delta_{ij},
\qquad
\mathcal P[\xi]\propto
\exp\!\left[-\frac1{2\Lambda}\int\xi(t)^2\dd t\right].
\]
\end{mytikzbox}

\begin{mytikzbox}{Multivariate Gaussian identities}{0.475}
For symmetric positive $A$ and mean zero,
\[
p(\bm x)=\frac{\sqrt{\det A}}{(2\pi)^{n/2}}
\exp\!\left(-\frac12\bm x^TA\bm x\right),
\quad C=A^{-1}.
\]
\[
I(\bm b)=\avg{\e^{\bm b^T\bm X}}
=\exp\!\left(\frac12\bm b^TC\bm b\right),
\quad \avg{X_iX_j}=C_{ij}.
\]
All odd centered moments vanish. Wick/Isserlis theorem:
\[
\avg{X_1\cdots X_{2n}}
=\sum_{\pi\in\mathcal P_{2n}}
\prod_{(i,j)\in\pi}C_{ij},
\qquad
\avg{X_1\cdots X_{2n+1}}=0,
\]
where \(\mathcal P_{2n}\) is the set of perfect matchings of \(\{1,\ldots,2n\}\)
(there are $(2n-1)!!=(2n)!/(2^n n!)$ of them). E.g.\
\[
\avg{X_1X_2X_3X_4}=C_{12}C_{34}+C_{13}C_{24}+C_{14}C_{23}.
\]
For Gaussian variables, zero covariance implies independence.
\end{mytikzbox} 

\begin{mytikzbox}{Langevin equations and moments}{0.475}
Underdamped Brownian particle:
\[
m\dot v=-\gamma v+F(x)+\xi(t),
\qquad \avg{\xi(t)\xi(t')}=2\gamma\kb T\delta(t-t').
\]
For $F=0$ (Ornstein--Uhlenbeck), $\tau_B=m/\gamma$ and
\[
v(t)=v_0\e^{-t/\tau_B}+\frac1m\int_0^t
\e^{-(t-s)/\tau_B}\xi(s)\dd s,
\]
\[
\avg v=v_0\e^{-t/\tau_B},
\quad
\Var(v)=\frac{\kb T}{m}\left(1-\e^{-2t/\tau_B}\right).
\]
Long times: $\avg{[x(t)-x(0)]^2}\simeq2Dt$ in one dimension,
$D=\kb T/\gamma$.

Overdamped form:
\[
\dd x=a(x,t)\dd t+b(x,t)\dd W_t.
\]
For It\^o calculus,
\[
\frac{\dd}{\dd t}\avg{A(x)}=
\avg{aA'+\frac12b^2A''}.
\]
Taking $A=x,x^2,\ldots$ gives moment equations without solving the full PDF.
\end{mytikzbox}

\begin{mytikzbox}{Gaussian path integrals}{0.475}
A path integral is an ordinary multidimensional integral with one variable for
each time or spatial point:
\[
\int \mathcal{D}\phi
\sim
\lim_{N\to\infty}\int\prod_{i=1}^{N}\dd\phi_i.
\]

For an ordinary Gaussian integral,
\[
\int \dd^n x\,
\e^{-\frac{1}{2}\bm{x}^{T}A\bm{x}+\bm{J}^{T}\bm{x}}
=
\sqrt{\frac{(2\pi)^n}{\det A}}\,
\e^{\frac{1}{2}\bm{J}^{T}A^{-1}\bm{J}}.
\]
The functional analogue is
\[
\int \mathcal{D}\phi\,
\e^{-\frac{1}{2}\int \phi K\phi+\int J\phi}
\propto
(\det K)^{-1/2}
\e^{\frac{1}{2}\int J K^{-1}J}.
\]
Thus \(K^{-1}\) is the covariance, or Green's function, of the Gaussian field.

For the Langevin equation
\[
\dot{x}(t)=a(x,t)+\xi(t),
\qquad
\avg{\xi(t)\xi(t')}=\Lambda\delta(t-t'),
\]
a trajectory \(x(t)\) requires the noise realization
\[
\xi(t)=\dot{x}(t)-a(x,t).
\]
Since Gaussian white noise has probability
\[
\mathcal{P}[\xi]
\propto
\exp\left[
-\frac{1}{2\Lambda}\int \xi(t)^2\,\dd t
\right],
\]
the corresponding trajectory probability is
\[
\mathcal{P}[x]
\propto
\mathcal{J}[x]\,
\exp\left[
-\frac{1}{2\Lambda}
\int
\left(\dot{x}(t)-a(x,t)\right)^2
\dd t
\right].
\]
Trajectories close to the deterministic equation
\[
\dot{x}(t)=a(x,t)
\]
are therefore the most probable.

A functional delta enforces an equation at every time:
\[
\delta[F]
\propto
\int \mathcal{D}\hat{x}\,
\exp\left[
i\int \hat{x}(t)F(t)\,\dd t
\right].
\]
For example, choosing
\[
F(t)=\dot{x}(t)-a(x,t)-\xi(t)
\]
imposes the Langevin equation inside the path integral.

The Jacobian \(\mathcal{J}[x]\) depends on the discretization convention and is
often constant for additive or linear noise.
\end{mytikzbox}

\begin{mytikzbox}{Kramers--Moyal expansion}{0.475}
Suppose that during a short time interval \(\tau\), a stochastic variable changes by
\[
\Delta x=x(t+\tau)-x(t),
\]
given that it started at \(x(t)=x\).

The short-time motion can be characterized by its moments:
\[
\avg{\Delta x\mid x(t)=x},
\qquad
\avg{(\Delta x)^2\mid x(t)=x},
\qquad
\avg{(\Delta x)^3\mid x(t)=x},
\ldots
\]

The Kramers--Moyal coefficients measure how quickly these moments grow:
\[
D^{(n)}(x,t)
=
\lim_{\tau\to0}
\frac{1}{n!\tau}
\avg{(\Delta x)^n\mid x(t)=x}.
\]

In particular,
\[
D^{(1)}
=
\lim_{\tau\to0}\frac{\avg{\Delta x}}{\tau}
\]
is the local average velocity, or drift, while
\[
D^{(2)}
=
\lim_{\tau\to0}\frac{\avg{(\Delta x)^2}}{2\tau}
\]
measures the local spreading due to randomness.

These short-time moments determine the evolution of the probability density:
\[
\partial_tP(x,t)
=
\sum_{n=1}^{\infty}
(-\partial_x)^n
\left[D^{(n)}(x,t)P(x,t)\right].
\]
If \(D^{(n)}=0\) for \(n\geq3\), this truncates to the \textbf{Fokker--Planck equation}
\[
\partial_tP
=
-\partial_x\bigl(D^{(1)}P\bigr)
+
\partial_x^2\bigl(D^{(2)}P\bigr).
\]

For the It\^o stochastic equation
\[
\dd x=a(x,t)\dd t+b(x,t)\dd W,
\]
the short-time change is
\[
\Delta x\simeq a\tau+b\Delta W,
\]
with
\[
\avg{\Delta W}=0,
\qquad
\avg{(\Delta W)^2}=\tau.
\]
Therefore,
\[
\avg{\Delta x}\simeq a\tau,
\qquad
\avg{(\Delta x)^2}\simeq b^2\tau,
\]
and all higher moments vanish after division by \(\tau\) in the limit
\(\tau\to0\). Hence,
\[
D^{(1)}=a,
\qquad
D^{(2)}=\frac{1}{2}b^2,
\qquad
D^{(n)}=0
\quad\text{for }n\geq3.
\]

Substituting these coefficients recovers
\[
\partial_tP
=
-\partial_x(aP)
+
\frac{1}{2}\partial_x^2(b^2P).
\]

Thus, \(a\) transports probability, while \(b\) causes probability to spread.

% \textbf{Pawula theorem:}
% A consistent finite truncation can contain either only the drift term, or the drift
% and diffusion terms. One cannot consistently stop at third or fourth order while
% preserving \(P(x,t)\geq0\).
\end{mytikzbox}

\begin{mytikzbox}{Fokker--Planck equation and probability current}{0.475}
For It\^o $\dd x=a(x,t)\dd t+b(x,t)\dd W_t$,
\[
\partial_tP=-\partial_x[aP]+\frac12\partial_x^2[b^2P]
=-\partial_xJ,
\]
\[
J=aP-\frac12\partial_x(b^2P).
\]
In $d$ dimensions, with $\dd x_i=a_i\dd t+B_{i\alpha}\dd W_\alpha$ and
$D_{ij}=(BB^T)_{ij}$,
\[
\partial_tP=-\partial_i(a_iP)+\frac12\partial_i\partial_j(D_{ij}P).
\]
Observable evolution (boundary terms vanishing):
\[
\frac{\dd}{\dd t}\avg{A(\bm x)}
=\avg{a_i\partial_iA+\tfrac12 D_{ij}\partial_i\partial_jA}.
\]
In 1D: $\frac{\dd}{\dd t}\avg x=\avg{a(x,t)}$ and
$\frac{\dd}{\dd t}\avg{x^2}=2\avg{xa}+\avg{b^2}$.
Boundary conditions: reflecting $J\cdot\hat n=0$; absorbing $P=0$.
Stationarity requires $\partial_xJ=0$; equilibrium with no net current has $J=0$.
For additive overdamped motion
\[
\dd x=\frac{F(x)}\gamma\dd t+\sqrt{2D}\dd W,
\qquad D=\frac{\kb T}{\gamma},
\]
$F=-U'$ gives
\[
P_{\rm eq}(x)=Z^{-1}\e^{-\beta U(x)}.
\]
\end{mytikzbox}

\begin{mytikzbox}{Ornstein--Uhlenbeck process}{0.475}
Free underdamped velocity ($F=0$), $\tau=m/\gamma$:
\[
\dd v=-\frac v\tau\dd t+\sqrt{2D_v}\,\dd W_t,
\quad
v(t)=v_0\e^{-t/\tau}
+\sqrt{2D_v}\int_0^t\e^{-(t-s)/\tau}\dd W_s,
\]
\[
\avg{v(t)}=v_0\e^{-t/\tau},
\qquad
\Var[v(t)]=D_v\tau\bigl(1-\e^{-2t/\tau}\bigr).
\]
Thermal: $D_v=\gamma\kb T/m^2=\kb T/(m\tau)$, so
\[
\avg{v(t)v(t')}_{\rm eq}
=\frac{\kb T}{m}\e^{-|t-t'|/\tau}.
\]
With $x(t)-x(0)=\int_0^t v(s)\dd s$ and $D=\kb T/\gamma$,
\[
\avg{[x(t)-x(0)]^2}
=2D\bigl[t-\tau(1-\e^{-t/\tau})\bigr],
\]
so $\avg{\Delta x^2}\simeq(\kb T/m)t^2$ for $t\ll\tau$ and
$\simeq2Dt$ for $t\gg\tau$.
\end{mytikzbox}

\begin{mytikzbox}{Klein--Kramers equation}{0.475}
For
\[
\dot x=v,
\qquad m\dot v=-\gamma v-U'(x)+\xi(t),
\qquad \avg{\xi(t)\xi(t')}=2\gamma\kb T\delta(t-t'),
\]
\[
\partial_tP=-v\partial_xP+\frac1mU'(x)\partial_vP
+\frac\gamma m\partial_v(vP)
+\frac{\gamma\kb T}{m^2}\partial_v^2P.
\]
The stationary solution is Maxwell--Boltzmann:
\[
P_{\rm eq}(x,v)=Z^{-1}
\exp\!\left[-\beta\left(\frac12mv^2+U(x)\right)\right].
\]
In the strongly overdamped limit, eliminate $v$ to obtain the Smoluchowski
Fokker--Planck equation for $x$.
\end{mytikzbox}

\begin{mytikzbox}{Continuous-time master equation}{0.475}
Let $W_{ij}$ be the transition rate $j\to i$. Then
\[
\dot P_i(t)=\sum_{j\ne i}\left[W_{ij}P_j-W_{ji}P_i\right]
=\sum_jL_{ij}P_j,
\]
where $L_{ij}=W_{ij}$ for $i\ne j$ and
$L_{ii}=-\sum_{k\ne i}W_{ki}$, so $\sum_iL_{ij}=0$.
A stationary state satisfies $L\bm P^{\rm st}=0$.
The zero eigenvalue is the stationary mode; the nonzero eigenvalues determine
relaxation times. Multiple closed communicating classes may give multiple
stationary states. Probability conservation follows by summing the master equation
over $i$.
\end{mytikzbox}

\begin{mytikzbox}{Probability-generating functions (master eq.)}{0.475}
For discrete $n=0,1,\ldots$,
\[
G(z,t)=\sum_{n=0}^\infty P_n(t)z^n,
\quad
G(1,t)=1,
\quad
\avg n=\partial_zG(1,t),
\]
\[
\avg{n(n-1)\cdots(n-k+1)}=\partial_z^kG(1,t).
\]
Useful translations:
\begin{gather*}
\sum_n nP_nz^n=z\partial_zG,
\quad
\sum_n(n+1)P_{n+1}z^n=\partial_zG,\\
\sum_n(n-1)P_{n-1}z^n=z^2\partial_zG.
\end{gather*}
Many birth--death master equations become first-order PDEs for $G$.
\end{mytikzbox}

\begin{mytikzbox}{Detailed balance}{0.475}
\textbf{Equilibrium condition} (pairwise cancellation) w.r.t.\ $P^{\rm eq}$:
\[
W_{ij}P_j^{\rm eq}=W_{ji}P_i^{\rm eq}
\qquad\forall i,j.
\]
It is sufficient for stationarity but stronger than stationarity: a stationary
nonequilibrium state may have circulating probability currents.
For the canonical distribution,
\[
\frac{W_{ij}}{W_{ji}}=\frac{P_i^{\rm eq}}{P_j^{\rm eq}}
=\e^{-\beta(E_i-E_j)}.
\]
Kolmogorov cycle criterion (for positive rates): detailed balance holds iff the
product of rates around every closed cycle equals the product around the reversed
cycle.
\end{mytikzbox}

\begin{mytikzbox}{Metropolis and Metropolis--Hastings}{0.475}
\textbf{Algorithm.} To sample a target $\pi_i$:
\begin{enumerate}
\item From state $i$, propose $j$ using $q(i\to j)$.
\item Accept with
\[
A(i\to j)=\min\!\left[1,
\frac{\pi_jq(j\to i)}{\pi_iq(i\to j)}\right].
\]
Otherwise remain at $i$.
\item Discard burn-in; estimate observables from the correlated chain and account
for autocorrelation.
\end{enumerate}
For \textbf{symmetric proposals} ($q(i\to j)=q(j\to i)$) and
$\pi_i\propto\e^{-\beta E_i}$,
\[
A(i\to j)=\min\!\left[1,\e^{-\beta(E_j-E_i)}\right].
\]
This construction obeys detailed balance. To converge to a unique target, the chain
must be irreducible and aperiodic. Smaller proposals accept often but mix slowly;
very large proposals are usually rejected.
\end{mytikzbox}
